1//! Core Modbus protocol definitions and enumerations.
2//!
3//! Contains the fundamental Modbus protocol definitions including function codes,
4//! exception codes, and diagnostic sub-function codes as specified in the
5//! Modbus Application Protocol Specification V1.1b3.
6//!
7//! Some enums intentionally cover more of the Modbus specification than the
8//! current command dispatcher implements. When an enum variant is recognized but
9//! not yet handled by the server, the crate returns
10//! [`ExceptionCode::IllegalFunction`].
11
12use crate::error::ModbusError;
13
14/// Modbus function codes as defined in the Modbus specification.
15///
16/// Each function code corresponds to a specific operation that can be performed
17/// on Modbus data. The error bit (0x80) is handled separately during frame parsing.
18///
19/// # Examples
20/// ```
21/// use libmodbuzz::FunctionCode;
22///
23/// let code = FunctionCode::ReadCoils;
24/// assert_eq!(u8::from(code), 0x01);
25/// ```
26#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
27#[repr(u8)]
28pub enum FunctionCode {
29    /// Read coils (discrete outputs) — FC 01
30    ReadCoils = 0x01,
31    /// Read discrete inputs — FC 02
32    ReadDiscreteInputs = 0x02,
33    /// Read holding registers — FC 03
34    ReadHoldingRegisters = 0x03,
35    /// Read input registers — FC 04
36    ReadInputRegisters = 0x04,
37    /// Write single coil — FC 05
38    WriteSingleCoil = 0x05,
39    /// Write single register — FC 06
40    WriteSingleRegister = 0x06,
41    /// Read exception status — FC 07.
42    ///
43    /// This function code is defined for protocol completeness, but the current
44    /// command dispatcher does not implement it yet.
45    ReadExceptionStatus = 0x07,
46    /// Diagnostics — FC 08
47    Diagnostics = 0x08,
48    /// Write multiple coils — FC 15 (0x0F)
49    WriteMultipleCoils = 0x0F,
50    /// Write multiple registers — FC 16 (0x10)
51    WriteMultipleRegisters = 0x10,
52}
53
54impl From<FunctionCode> for u8 {
55    fn from(code: FunctionCode) -> u8 {
56        code as u8
57    }
58}
59
60impl TryFrom<u8> for FunctionCode {
61    type Error = ModbusError;
62
63    fn try_from(value: u8) -> Result<Self, Self::Error> {
64        match value {
65            0x01 => Ok(FunctionCode::ReadCoils),
66            0x02 => Ok(FunctionCode::ReadDiscreteInputs),
67            0x03 => Ok(FunctionCode::ReadHoldingRegisters),
68            0x04 => Ok(FunctionCode::ReadInputRegisters),
69            0x05 => Ok(FunctionCode::WriteSingleCoil),
70            0x06 => Ok(FunctionCode::WriteSingleRegister),
71            0x07 => Ok(FunctionCode::ReadExceptionStatus),
72            0x08 => Ok(FunctionCode::Diagnostics),
73            0x0F => Ok(FunctionCode::WriteMultipleCoils),
74            0x10 => Ok(FunctionCode::WriteMultipleRegisters),
75            _ => Err(ModbusError::UnsupportedFunctionCode(value)),
76        }
77    }
78}

80/// Modbus exception codes returned in error responses.
81///
82/// When a Modbus request cannot be processed successfully, the server responds
83/// with an exception frame containing one of these codes to indicate the
84/// specific error condition.
85#[derive(Debug, Clone, Copy, PartialEq, Eq)]
86#[repr(u8)]
87pub enum ExceptionCode {
88    /// Function code not supported by the device
89    IllegalFunction = 0x01,
90    /// Data address not valid for the device
91    IllegalDataAddress = 0x02,
92    /// Data value not valid for the request
93    IllegalDataValue = 0x03,
94    /// Unrecoverable error in slave device
95    SlaveDeviceFailure = 0x04,
96    /// Request accepted, processing (long duration)
97    Acknowledge = 0x05,
98    /// Slave device is busy processing another request
99    SlaveDeviceBusy = 0x06,
100    /// Request cannot be performed (programming failures)
101    NegativeAcknowledgment = 0x07,
102    /// Memory parity error detected
103    MemoryParityError = 0x08,
104    /// Gateway path unavailable
105    GatewayPathUnavailable = 0x0A,
106    /// Gateway target device failed to respond
107    GatewayTargetDeviceFailedToRespond = 0x0B,
108}
109
110impl From<ExceptionCode> for u8 {
111    fn from(code: ExceptionCode) -> u8 {
112        code as u8
113    }
114}
115
116impl TryFrom<u8> for ExceptionCode {
117    type Error = ModbusError;
118
119    fn try_from(value: u8) -> Result<Self, Self::Error> {
120        match value {
121            0x01 => Ok(ExceptionCode::IllegalFunction),
122            0x02 => Ok(ExceptionCode::IllegalDataAddress),
123            0x03 => Ok(ExceptionCode::IllegalDataValue),
124            0x04 => Ok(ExceptionCode::SlaveDeviceFailure),
125            0x05 => Ok(ExceptionCode::Acknowledge),
126            0x06 => Ok(ExceptionCode::SlaveDeviceBusy),
127            0x07 => Ok(ExceptionCode::NegativeAcknowledgment),
128            0x08 => Ok(ExceptionCode::MemoryParityError),
129            0x0A => Ok(ExceptionCode::GatewayPathUnavailable),
130            0x0B => Ok(ExceptionCode::GatewayTargetDeviceFailedToRespond),
131            _ => Err(ModbusError::InvalidExceptionCode(value)),
132        }
133    }
134}

136/// Diagnostic sub-function codes for function code 08 (Diagnostics).
137///
138/// The diagnostics function (FC 08) supports various sub-functions for testing
139/// communication and retrieving diagnostic information from Modbus devices.
140/// The current implementation handles a focused subset; unsupported codes
141/// return [`ModbusError::UnsupportedDiagnosticsCode`].
142#[derive(Debug, Clone, Copy, PartialEq, Eq)]
143#[repr(u16)]
144pub enum DiagnosticsCode {
145    /// Echo back query data (loopback test)
146    ReturnQueryData = 0x0000,
147    /// Restart communications option
148    RestartCommunicationsOption = 0x0001,
149    /// Return diagnostic register contents
150    ReturnDiagnosticRegister = 0x0002,
151    /// Change ASCII input delimiter
152    ChangeAsciiInputDelimiter = 0x0003,
153    /// Force listen only mode
154    ForceListenOnlyMode = 0x0004,
155    /// Clear all counters and diagnostic register
156    ClearCountersAndDiagnosticRegister = 0x000A,
157    /// Return bus message count
158    ReturnBusMessageCount = 0x000B,
159    /// Return bus communication error count
160    ReturnBusCommunicationErrorCount = 0x000C,
161    /// Return bus exception error count
162    ReturnBusExceptionErrorCount = 0x000D,
163    /// Return slave message count
164    ReturnSlaveMessageCount = 0x000E,
165    /// Return slave no response count
166    ReturnSlaveNoResponseCount = 0x000F,
167    /// Return slave NAK count
168    ReturnSlaveNakCount = 0x0010,
169    /// Return slave busy count
170    ReturnSlaveBusyCount = 0x0011,
171    /// Return bus character overrun count
172    ReturnBusCharacterOverrunCount = 0x0012,
173}
174
175impl From<DiagnosticsCode> for u16 {
176    fn from(code: DiagnosticsCode) -> u16 {
177        code as u16
178    }
179}
180
181impl TryFrom<u16> for DiagnosticsCode {
182    type Error = ModbusError;
183
184    fn try_from(value: u16) -> Result<Self, Self::Error> {
185        match value {
186            0x0000 => Ok(DiagnosticsCode::ReturnQueryData),
187            0x0001 => Ok(DiagnosticsCode::RestartCommunicationsOption),
188            0x0002 => Ok(DiagnosticsCode::ReturnDiagnosticRegister),
189            0x0003 => Ok(DiagnosticsCode::ChangeAsciiInputDelimiter),
190            0x0004 => Ok(DiagnosticsCode::ForceListenOnlyMode),
191            0x000A => Ok(DiagnosticsCode::ClearCountersAndDiagnosticRegister),
192            0x000B => Ok(DiagnosticsCode::ReturnBusMessageCount),
193            0x000C => Ok(DiagnosticsCode::ReturnBusCommunicationErrorCount),
194            0x000D => Ok(DiagnosticsCode::ReturnBusExceptionErrorCount),
195            0x000E => Ok(DiagnosticsCode::ReturnSlaveMessageCount),
196            0x000F => Ok(DiagnosticsCode::ReturnSlaveNoResponseCount),
197            0x0010 => Ok(DiagnosticsCode::ReturnSlaveNakCount),
198            0x0011 => Ok(DiagnosticsCode::ReturnSlaveBusyCount),
199            0x0012 => Ok(DiagnosticsCode::ReturnBusCharacterOverrunCount),
200            _ => Err(ModbusError::UnsupportedDiagnosticsCode(value)),
201        }
202    }
203}

205#[cfg(test)]
206mod tests {
207    use super::*;
208
209    #[test]
210    fn function_code_round_trip() {
211        for code in [
212            FunctionCode::ReadCoils,
213            FunctionCode::ReadDiscreteInputs,
214            FunctionCode::ReadHoldingRegisters,
215            FunctionCode::ReadInputRegisters,
216            FunctionCode::WriteSingleCoil,
217            FunctionCode::WriteSingleRegister,
218            FunctionCode::Diagnostics,
219            FunctionCode::WriteMultipleCoils,
220            FunctionCode::WriteMultipleRegisters,
221        ] {
222            let byte = u8::from(code);
223            let parsed = FunctionCode::try_from(byte).unwrap();
224            assert_eq!(code, parsed);
225        }
226    }
227
228    #[test]
229    fn invalid_function_code() {
230        assert!(FunctionCode::try_from(0xFF).is_err());
231        assert!(FunctionCode::try_from(0x00).is_err());
232    }
233
234    #[test]
235    fn exception_code_round_trip() {
236        let code = ExceptionCode::IllegalFunction;
237        assert_eq!(u8::from(code), 0x01);
238        assert_eq!(ExceptionCode::try_from(0x01).unwrap(), code);
239    }
240
241    #[test]
242    fn diagnostics_code_round_trip() {
243        let code = DiagnosticsCode::ReturnQueryData;
244        assert_eq!(u16::from(code), 0x0000);
245        assert_eq!(DiagnosticsCode::try_from(0x0000).unwrap(), code);
246    }
