1//! Modbus TCP server implementation.
2//!
3//! Handles the Modbus TCP request processing pipeline:
4//! 1. MBAP header parsing
5//! 2. Unit ID (slave address) matching
6//! 3. Function code dispatch
7//! 4. Response frame generation with MBAP header
8
9use crate::command;
10use crate::data_model::DataModel;
11use crate::frame::tcp::TcpFrame;
12use crate::server::ModbusServer;
13
14/// Modbus TCP server with configurable unit ID.
15///
16/// Unlike RTU, TCP frames use the MBAP header instead of CRC for framing.
17/// The `unit_id` field in the MBAP header serves the same purpose as the
18/// slave address in RTU. A server configured with unit ID `0` accepts requests
19/// for any incoming Unit ID and echoes that Unit ID back in the response.
20///
21/// # Examples
22/// ```
23/// use libmodbuzz::server::tcp::TcpServer;
24/// use libmodbuzz::server::ModbusServer;
25///
26/// let mut server = TcpServer::new(1);
27/// server.data.write_holding_register(0, 0xCAFE);
28///
29/// // Build a Modbus TCP Read Holding Registers request
30/// // MBAP: txn=1, proto=0, len=6, unit=1 | FC=03, addr=0x0000, qty=0x0001
31/// let request = vec![
32///     0x00, 0x01, 0x00, 0x00, 0x00, 0x06, 0x01, // MBAP header
33///     0x03, 0x00, 0x00, 0x00, 0x01,              // PDU
34/// ];
35/// let response = server.process(&request).unwrap();
36/// assert!(response.len() > 7); // MBAP header + PDU
37/// ```
38#[derive(Debug)]
39pub struct TcpServer {
40    /// The unit ID this server responds to (1–247, or 0 for any).
41    pub unit_id: u8,
42    /// The Modbus data model holding coils, inputs, and registers.
43    pub data: DataModel,
44}
45
46impl TcpServer {
47    /// Create a new TCP server with the given unit ID and default data model.
48    pub fn new(unit_id: u8) -> Self {
49        Self {
50            unit_id,
51            data: DataModel::default(),
52        }
53    }
54
55    /// Create a new TCP server with the given unit ID and custom data model.
56    pub fn with_data_model(unit_id: u8, data: DataModel) -> Self {
57        Self { unit_id, data }
58    }
59}
60
61impl ModbusServer for TcpServer {
62    fn process(&mut self, request_data: &[u8]) -> Option<Vec<u8>> {
63        // Parse the TCP frame (MBAP header + PDU)
64        let request = match TcpFrame::deserialize(request_data) {
65            Ok(frame) => frame,
66            Err(_) => return None,
67        };
68
69        // Check unit ID — respond only to our unit ID (or 0 = any)
70        if self.unit_id != 0 && request.header.unit_id != self.unit_id {
71            return None;
72        }
73
74        // Execute the command
75        let response_pdu = command::execute(&mut self.data, &request.pdu);
76
77        // Build response TCP frame, preserving the transaction ID from the request
78        let response_frame = TcpFrame::new(
79            request.header.transaction_id,
80            request.header.unit_id,
81            response_pdu,
82        );
83
84        Some(response_frame.serialize())
85    }
86}
