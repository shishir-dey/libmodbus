1//! Modbus TCP frame serialization and deserialization.
2//!
3//! TCP frames use the MBAP (Modbus Application Protocol) header followed by the
4//! PDU. No checksum is needed since TCP provides reliable delivery.
5//! Unlike RTU, TCP framing does not encode an explicit request/response role,
6//! so callers provide that context when interpreting parsed frames.
7//!
8//! # Frame Layout
9//! ```text
10//! | Transaction ID (2) | Protocol ID (2) | Length (2) | Unit ID (1) | PDU (N) |
11//! ```
12
13use crate::error::{ModbusError, Result};
14use crate::frame::{FrameType, Pdu};
15use crate::protocol::{ExceptionCode, FunctionCode};
16
17/// MBAP (Modbus Application Protocol) header for TCP frames.
18///
19/// The 7-byte header provides transaction tracking, protocol identification,
20/// length information, and unit identification.
21#[derive(Debug, Clone, Copy, PartialEq, Eq)]
22pub struct MbapHeader {
23    /// Transaction identifier for matching requests to responses.
24    pub transaction_id: u16,
25    /// Protocol identifier (always 0x0000 for Modbus).
26    pub protocol_id: u16,
27    /// Number of following bytes (unit ID + PDU length).
28    pub length: u16,
29    /// Unit identifier (equivalent to slave address in RTU).
30    pub unit_id: u8,
31}
32
33impl MbapHeader {
34    /// MBAP header size in bytes.
35    pub const SIZE: usize = 7;
36
37    /// Create a new MBAP header.
38    ///
39    /// `pdu_length` is the number of bytes in the wire-format PDU, excluding
40    /// the Unit ID byte but including the function code and payload.
41    pub fn new(transaction_id: u16, unit_id: u8, pdu_length: u16) -> Self {
42        Self {
43            transaction_id,
44            protocol_id: 0x0000,
45            length: pdu_length + 1, // +1 for unit_id byte
46            unit_id,
47        }
48    }
49
50    /// Serialize the MBAP header to bytes (big-endian).
51    pub fn serialize(&self) -> [u8; 7] {
52        let mut buf = [0u8; 7];
53        buf[0] = (self.transaction_id >> 8) as u8;
54        buf[1] = (self.transaction_id & 0xFF) as u8;
55        buf[2] = (self.protocol_id >> 8) as u8;
56        buf[3] = (self.protocol_id & 0xFF) as u8;
57        buf[4] = (self.length >> 8) as u8;
58        buf[5] = (self.length & 0xFF) as u8;
59        buf[6] = self.unit_id;
60        buf
61    }
62
63    /// Deserialize an MBAP header from a byte slice.
64    ///
65    /// # Errors
66    /// Returns [`ModbusError::InvalidFrame`] if the slice is shorter than 7 bytes
67    /// or the protocol ID is not 0x0000.
68    pub fn deserialize(data: &[u8]) -> Result<Self> {
69        if data.len() < Self::SIZE {
70            return Err(ModbusError::InvalidFrame {
71                reason: format!("MBAP header requires {} bytes", Self::SIZE),
72                actual_len: data.len(),
73            });
74        }
75
76        let protocol_id = u16::from(data[2]) << 8 | u16::from(data[3]);
77        if protocol_id != 0x0000 {
78            return Err(ModbusError::InvalidFrame {
79                reason: format!(
80                    "invalid protocol ID: expected 0x0000, got 0x{:04X}",
81                    protocol_id
82                ),
83                actual_len: data.len(),
84            });
85        }
86
87        Ok(Self {
88            transaction_id: u16::from(data[0]) << 8 | u16::from(data[1]),
89            protocol_id,
90            length: u16::from(data[4]) << 8 | u16::from(data[5]),
91            unit_id: data[6],
92        })
93    }
94}

96/// A complete Modbus TCP frame (MBAP header + PDU).
97///
98/// # Examples
99/// ```
100/// use libmodbuzz::frame::tcp::TcpFrame;
101/// use libmodbuzz::frame::Pdu;
102/// use libmodbuzz::FunctionCode;
103///
104/// let pdu = Pdu::new_request(
105///     FunctionCode::ReadHoldingRegisters,
106///     vec![0x00, 0x00, 0x00, 0x0A],
107/// );
108/// let frame = TcpFrame::new(1, 0x01, pdu);
109/// let bytes = frame.serialize();
110/// assert_eq!(bytes.len(), 7 + 1 + 4); // MBAP + FC + data
111/// ```
112#[derive(Debug, Clone, PartialEq, Eq)]
113pub struct TcpFrame {
114    /// The MBAP header.
115    pub header: MbapHeader,
116    /// The protocol data unit.
117    pub pdu: Pdu,
118}
119
120impl TcpFrame {
121    /// Create a new TCP frame.
122    ///
123    /// Automatically calculates the MBAP `length` field from the PDU.
124    pub fn new(transaction_id: u16, unit_id: u8, pdu: Pdu) -> Self {
125        let pdu_wire_len = match pdu.frame_type {
126            FrameType::Exception => 2, // FC + exception code
127            _ => 1 + pdu.data.len(),   // FC + data
128        };
129
130        Self {
131            header: MbapHeader::new(transaction_id, unit_id, pdu_wire_len as u16),
132            pdu,
133        }
134    }
135
136    /// Serialize the TCP frame to a byte vector.
137    pub fn serialize(&self) -> Vec<u8> {
138        let mut output = Vec::with_capacity(MbapHeader::SIZE + 1 + self.pdu.data.len());
139        output.extend_from_slice(&self.header.serialize());
140
141        match self.pdu.frame_type {
142            FrameType::Exception => {
143                output.push(u8::from(self.pdu.function_code) | 0x80);
144                output.push(self.pdu.exception_code.map(u8::from).unwrap_or(0));
145            }
146            _ => {
147                output.push(u8::from(self.pdu.function_code));
148                output.extend_from_slice(&self.pdu.data);
149            }
150        }
151
152        output
153    }
154
155    /// Deserialize a byte slice into a TCP frame.
156    ///
157    /// # Errors
158    /// Returns [`ModbusError::InvalidFrame`] if the data is too short or malformed.
159    ///
160    /// Non-exception frames are returned with [`FrameType::Response`] because
161    /// the MBAP header does not distinguish requests from responses. Server-side
162    /// code should interpret the frame in the context in which it was received.
163    pub fn deserialize(data: &[u8]) -> Result<Self> {
164        let header = MbapHeader::deserialize(data)?;
165
166        let pdu_start = MbapHeader::SIZE;
167        if data.len() < pdu_start + 1 {
168            return Err(ModbusError::InvalidFrame {
169                reason: "TCP frame has no PDU after MBAP header".into(),
170                actual_len: data.len(),
171            });
172        }
173
174        let raw_fc = data[pdu_start];
175
176        if raw_fc & 0x80 != 0 {
177            // Exception response
178            let function_code = FunctionCode::try_from(raw_fc & 0x7F)?;
179            let exception_code = if data.len() > pdu_start + 1 {
180                Some(ExceptionCode::try_from(data[pdu_start + 1])?)
181            } else {
182                None
183            };
184
185            Ok(Self {
186                header,
187                pdu: Pdu {
188                    frame_type: FrameType::Exception,
189                    function_code,
190                    data: Vec::new(),
191                    exception_code,
192                },
193            })
194        } else {
195            let function_code = FunctionCode::try_from(raw_fc)?;
196            let payload = if data.len() > pdu_start + 1 {
197                data[pdu_start + 1..].to_vec()
198            } else {
199                Vec::new()
200            };
201
202            Ok(Self {
203                header,
204                pdu: Pdu {
205                    frame_type: FrameType::Response, // Inferred; caller can override
206                    function_code,
207                    data: payload,
208                    exception_code: None,
209                },
210            })
211        }
212    }
213}

215#[cfg(test)]
216mod tests {
217    use super::*;
218
219    #[test]
220    fn mbap_header_round_trip() {
221        let header = MbapHeader::new(0x0001, 0x01, 6);
222        let bytes = header.serialize();
223        let parsed = MbapHeader::deserialize(&bytes).unwrap();
224        assert_eq!(header.transaction_id, parsed.transaction_id);
225        assert_eq!(header.unit_id, parsed.unit_id);
226        assert_eq!(header.length, parsed.length);
227    }
228
229    #[test]
230    fn tcp_frame_round_trip() {
231        let pdu = Pdu::new_request(
232            FunctionCode::ReadHoldingRegisters,
233            vec![0x00, 0x00, 0x00, 0x0A],
234        );
235        let frame = TcpFrame::new(1, 0x01, pdu);
236        let bytes = frame.serialize();
237        let parsed = TcpFrame::deserialize(&bytes).unwrap();
238
239        assert_eq!(parsed.header.transaction_id, 1);
240        assert_eq!(parsed.header.unit_id, 0x01);
241        assert_eq!(parsed.pdu.function_code, FunctionCode::ReadHoldingRegisters);
242        assert_eq!(parsed.pdu.data, vec![0x00, 0x00, 0x00, 0x0A]);
243    }
244
245    #[test]
246    fn tcp_exception_round_trip() {
247        let pdu = Pdu::new_exception(FunctionCode::ReadCoils, ExceptionCode::IllegalDataAddress);
248        let frame = TcpFrame::new(42, 0x01, pdu);
249        let bytes = frame.serialize();
250        let parsed = TcpFrame::deserialize(&bytes).unwrap();
251
252        assert_eq!(parsed.pdu.frame_type, FrameType::Exception);
253        assert_eq!(parsed.pdu.function_code, FunctionCode::ReadCoils);
254        assert_eq!(
255            parsed.pdu.exception_code,
256            Some(ExceptionCode::IllegalDataAddress)
257        );
258    }
259
260    #[test]
261    fn mbap_too_short() {
262        assert!(MbapHeader::deserialize(&[0x00, 0x01]).is_err());
263    }
264
265    #[test]
266    fn mbap_wrong_protocol_id() {
267        let mut bytes = MbapHeader::new(1, 1, 5).serialize();
268        bytes[2] = 0xFF; // corrupt protocol ID
269        assert!(MbapHeader::deserialize(&bytes).is_err());
270    }
